## 一、错误处理规范

### 1.1 错误类型划分

Rust 中的错误分为两大类，你们需要掌握：

```rust
// ═══════════════════════════════════════════════════════════════════════
// 可恢复错误 (Recoverable) - 使用 Result<T, E>
// 适用场景：业务逻辑错误、验证失败、资源未找到等
// ═══════════════════════════════════════════════════════════════════════

#[derive(Error, Debug)]
pub enum AppError {
    #[error("日程不存在: {0}")]
    EventNotFound(String),

    #[error("用户不存在: {0}")]
    UserNotFound(String),

    #[error("无效的API Key")]
    InvalidApiKey,

    #[error("权限不足")]
    InsufficientPermissions,

    #[error("参数验证失败: {0}")]
    ValidationError(String),
}

// ═══════════════════════════════════════════════════════════════════════
// 不可恢复错误 (Unrecoverable) - 使用 panic!
// 适用场景：程序逻辑错误、严重系统故障
// 注意：绝不要在库代码中 panic!
// ═══════════════════════════════════════════════════════════════════════

fn critical_operation() {
    // 如果发生这种情况，说明代码有严重bug
    unimplemented!("此功能尚未实现");
}
```

### 1.3 错误处理链式调用

```rust
// ✅ 推荐：使用 ? 操作符进行错误传播
pub async fn get_event(&self, id: &str) -> AppResult<Event> {
    let event = sqlx::query_as::<_, Event>("SELECT * FROM events WHERE id = ?1")
        .bind(id)
        .fetch_one(&self.pool)
        .await
        .map_err(|_| AppError::EventNotFound(id.to_string()))?;

    Ok(event)
}

// ❌ 错误：过度使用 match 而不利用 ? 操作符
pub async fn bad_example(&self, id: &str) -> AppResult<Event> {
    match sqlx::query_as::<_, Event>("SELECT * FROM events WHERE id = ?1")
        .bind(id)
        .fetch_one(&self.pool)
        .await
    {
        Ok(event) => Ok(event),
        Err(_) => Err(AppError::EventNotFound(id.to_string())),
    }
}

// ❌ 错误：使用 unwrap() 而不处理错误
pub async fn terrible_example(&self, id: &str) -> Event {
    sqlx::query_as::<_, Event>("SELECT * FROM events WHERE id = ?1")
        .bind(id)
        .fetch_one(&self.pool)
        .await
        .unwrap() // 如果数据库出错，整个程序会 panic!
}
```

## 二、数据库操作规范 (SQLx)

### 2.1 连接池管理

```rust
// ✅ 正确的连接池创建模式
pub async fn create_pool(database_url: &str) -> anyhow::Result<SqlitePool> {
    // 1. 确保数据目录存在
    if let Some(parent) = std::path::Path::new(database_url).strip_prefix("sqlite:") {
        std::fs::create_dir_all(parent)?;
    }

    // 2. 创建连接池
    let pool = SqlitePool::connect(database_url).await?;

    // 3. 启用外键约束
    sqlx::query("PRAGMA foreign_keys = ON")
        .execute(&pool)
        .await?;

    Ok(pool)
}
```

### 2.2 事务处理

```rust
// ✅ 使用事务保证数据一致性
pub async fn create_event_with_user_check(
    &self,
    user_id: &str,
    input: CreateEvent
) -> AppResult<Event> {
    let mut tx = self.pool.begin().await?;

    // 1. 验证用户存在
    let user_exists = sqlx::query_scalar::<_, bool>(
        "SELECT EXISTS(SELECT 1 FROM users WHERE id = ?1)"
    )
    .bind(user_id)
    .fetch_one(&mut *tx)
    .await?;

    if !user_exists {
        return Err(AppError::UserNotFound(user_id.to_string()));
    }

    // 2. 创建日程
    let event = Event::new(user_id.to_string(), input);
    sqlx::query(
        "INSERT INTO events (id, user_id, title, ...) VALUES (?1, ?2, ?3, ...)"
    )
    .bind(&event.id)
    .bind(&event.user_id)
    .bind(&event.title)
    .execute(&mut *tx)
    .await?;

    // 3. 提交事务
    tx.commit().await?;

    Ok(event)
}
```

### 2.3 查询参数绑定

```rust
// ✅ 使用动态参数构建安全的 SQL
pub async fn search_events(&self, query: &EventQuery) -> AppResult<Vec<Event>> {
    let mut sql = String::from("SELECT * FROM events WHERE 1=1");
    let mut params: Vec<String> = Vec::new();

    if let Some(status) = &query.status {
        sql.push_str(&format!(" AND status = ${}", params.len() + 1));
        params.push(status.clone());
    }

    if let Some(keyword) = &query.keyword {
        sql.push_str(&format!(
            " AND (title LIKE ${} OR description LIKE ${})",
            params.len() + 1,
            params.len() + 2
        ));
        params.push(format!("%{}%", keyword));
        params.push(format!("%{}%", keyword));
    }

    // 构建并执行查询
    let mut q = sqlx::query_as::<_, Event>(&sql);
    for param in params {
        q = q.bind(param);
    }

    q.fetch_all(&self.pool).await.map_err(AppError::Database)
}
```

## 四、认证与授权

### 4.1 API Key 认证中间件

```rust
// src/middleware/auth.rs

pub struct AuthenticatedUser {
    pub user_id: String,
    pub is_admin: bool,
}

// 认证中间件
pub async fn auth_middleware(
    State(user_repo): State<Arc<UserRepository>>,
    headers: HeaderMap,
    mut request: Request,
    next: Next,
) -> Result<Response, AppError> {
    // 1. 从 Header 获取 API Key
    let api_key = headers
        .get("X-API-Key")
        .and_then(|v| v.to_str().ok())
        .ok_or(AppError::InvalidApiKey)?;

    // 2. 验证 API Key
    let user = user_repo.find_by_api_key(api_key).await?;

    // 3. 将用户信息注入请求扩展
    request.extensions_mut().insert(AuthenticatedUser {
        user_id: user.id,
        is_admin: user.is_admin,
    });

    // 4. 继续处理请求
    Ok(next.run(request).await)
}

// 权限检查辅助函数
pub fn require_admin(auth: &AuthenticatedUser) -> AppResult<()> {
    if !auth.is_admin {
        return Err(AppError::InsufficientPermissions);
    }
    Ok(())
}

pub fn check_ownership(auth: &AuthenticatedUser, resource_user_id: &str) -> AppResult<()> {
    if !auth.is_admin && auth.user_id != resource_user_id {
        return Err(AppError::InsufficientPermissions);
    }
    Ok(())
}
```

### 9.3 API 限流 (Rate Limiting)

```rust
use std::time::Duration;
use std::collections::HashMap;
use tokio::sync::Mutex;

pub struct RateLimiter {
    requests: Arc<Mutex<HashMap<String, Vec<std::time::Instant>>>>,
    max_requests: usize,
    window: Duration,
}

impl RateLimiter {
    pub async fn check(&self, key: &str) -> bool {
        let mut requests = self.requests.lock().await;
        let now = std::time::Instant::now();

        // 清理过期记录
        requests.entry(key.to_string())
            .or_insert_with(Vec::new);

        let times = requests.get_mut(key).unwrap();

        // 移除窗口外的请求
        times.retain(|&t| now.duration_since(t) < self.window);

        if times.len() >= self.max_requests {
            return false;
        }

        times.push(now);
        true
    }
}
```
